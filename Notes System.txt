-- SQL Notes -------------------------------------------------------------------

Create a full history system, a la Wikipedia.

SQL DD:
	Location_ID (primary key) - a key to add to the Mu* dbref to identify them
	editor's dbref (primary key)
	timestamp (primary key)
	target's creation date (not null)
		- convtime(get(me/created), , 6)
		- target dbref + creation date = UUID
	target'd debref (not null, to validate vs game dbref + _location_ID)
	the complete new note, even if only edited (not null)
	permissions (default: private)
	approval dbref

--

CREATE TABLE [edit(mudname(), %b, )]_notes
(
    target           INT             NOT NULL, 
    editor           INT             NOT NULL, 
    time             TIMESTAMP       NOT NULL, 
    target_created   INT             NOT NULL, 
    title            VARCHAR(255),   NOT NULL, 
    note             TEXT            NOT NULL, 
    permission       VARCHAR(255)    DEFAULT 'private', 
    approver         INT, 

    PRIMARY KEY (target, editor, time), 
    INDEX target_idx (target, target_created)
);
